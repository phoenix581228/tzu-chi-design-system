🔧 CSS變數定義

/* CI系統透明度變數 */
:root {
  /* On Primary 透明度 - 在主色 #87CAFE 上使用 */
  --on-primary-high: rgba(0, 0, 0, 1.0);      /* 100% 不透明度 */
  --on-primary-medium: rgba(0, 0, 0, 0.74);   /* 74% 不透明度 */
  --on-primary-disabled: rgba(0, 0, 0, 0.38); /* 38% 不透明度 */
  
  /* On Surface 透明度 - 在表面色 #121212 上使用 */
  --on-surface-high: rgba(255, 255, 255, 1.0);    /* 100% 不透明度 */
  --on-surface-medium: rgba(255, 255, 255, 0.6);  /* 60% 不透明度 */
  --on-surface-disabled: rgba(255, 255, 255, 0.38); /* 38% 不透明度 */
}
